vector Insertion_sort(vector& A)
{
    for(vector::size_type i=1;i
    {
        int key=A[i];
        vector::size_type j=i-1;
        while(j>=0&&A[j]<key)
        {
            A[j+1]=A[j];
            j=j-1;
        }
        A[j+1]=key;
    }
    return A;
}
int main()
{

    int x;
    cout<<"Please enter some numbers:";

    for(vector::size_type i=0;i<A.size();++i)
        cout<<A[i]<<" ";
    cout<<endl;
