int shift_circular_left(int x,int y){
    int t=a[x];
    for(int i=x;i<=y-1;++i){
        a[i]=a[i+1];
    }
    a[y]=t;
}
int shift_circular_right(int x,int y){
    int t=a[y];
    for(int i=y;i>=x+1;--i){
        a[i]=a[i-1];
    }
    a[x]=t;
}
int find(int x){
    for(int i=0;i<=m;++i){
        if(a[i]==x)return i;
