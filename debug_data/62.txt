int getline(char line[]){
    int i;
    int c;
    for(i = 0;(c = getchar()) != EOF&&c != '\n';i++)
        line[i] = c;
    if(c = '\n'){
        line[i] = c;
        i++;
    }
    line[i] = '\0';
    return i;
}
