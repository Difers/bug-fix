### 修改方案：
引用一楼，但是觉得应该稍微修改

```python
 
flag1=0
flag2=0
 
def run():
    global flag1,flag2
    a = input()
    b = input()
    c = input()
    d = input()
    condition1 = a>b and c<d
    condition2 = a<b and c>d
    if condition1 and flag1==0:
        print('condition1')
        flag1=1
        flag2=0
    elif condition2 and flag2==0:
        print('conditon2')
        flag2=1
        flag1=0
    else:
        print('keep')
        
        
schedule.every(1).minutes.do(run)
 


```
